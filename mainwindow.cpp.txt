void MainWindow::setupUI() {
    QWidget *centralWidget = new QWidget(this);
    centralWidget->setObjectName("centralWidget");   // ← TAMBAHKAN BARIS INI
    setCentralWidget(centralWidget);

    // Layout Utama (Horizontal membelah Kiri dan Kanan)
    QHBoxLayout *mainLayout = new QHBoxLayout(centralWidget);

    // --- PANEL KIRI: Daftar Pilihan Matkul ---
    QVBoxLayout *leftLayout = new QVBoxLayout();
    QLabel *lblMaster = new QLabel("Daftar Mata Kuliah Tersedia (Smt 3):", this);
    lblMaster->setObjectName("HeaderLabel");

    listMasterMatkul = new QListWidget(this);
    for (const auto& matkul : std::as_const(masterMatkul)) {
        QString waktuStr = matkul.jamMulai.toString("HH:mm") + " - " + matkul.jamSelesai().toString("HH:mm");

        // SOLUSI: Menggunakan multi-arg (memasukkan beberapa string sekaligus di satu fungsi .arg)
        // Sesuai dengan rekomendasi Clazy untuk performa alokasi memori yang lebih baik
        QString itemText = QString("[%1] %2 (%3 SKS)\n📅 %4, %5 | 📚 %6")
                               .arg(matkul.kode, matkul.nama, QString::number(matkul.sks), matkul.hari, waktuStr, matkul.ruang);

        listMasterMatkul->addItem(itemText);
    }

    btnTambah = new QPushButton(" Tambah ke Jadwal", this);
    leftLayout->addWidget(lblMaster);
    leftLayout->addWidget(listMasterMatkul);
    leftLayout->addWidget(btnTambah);

    // --- PANEL KANAN: Hasil Jadwal & Deteksi Bentrok ---
    QVBoxLayout *rightLayout = new QVBoxLayout();
    QLabel *lblJadwal = new QLabel("Jadwal Kuliah Saya:", this);
    lblJadwal->setObjectName("HeaderLabel");

    tableJadwal = new QTableWidget(this);
    tableJadwal->setColumnCount(6);
    tableJadwal->setHorizontalHeaderLabels({"Kode", "Mata Kuliah", "SKS", "Waktu", "Ruang", "Status"});
    // Ambil kontrol header tabel
    QHeaderView *header = tableJadwal->horizontalHeader();

    // 1. Atur agar kolom Kode (0), SKS (2), dan Ruang (4) ukurannya pas dengan isi teksnya
    header->setSectionResizeMode(0, QHeaderView::ResizeToContents);
    header->setSectionResizeMode(2, QHeaderView::ResizeToContents);
    header->setSectionResizeMode(4, QHeaderView::ResizeToContents);

    // 2. Atur agar kolom Mata Kuliah (1) mengambil sisa ruang paling besar yang tersedia
    header->setSectionResizeMode(1, QHeaderView::Stretch);

    // 3. Atur agar kolom Waktu (3) dan Status (5) otomatis menyesuaikan panjang teks jadwal
    header->setSectionResizeMode(3, QHeaderView::ResizeToContents);
    header->setSectionResizeMode(5, QHeaderView::ResizeToContents);
    tableJadwal->setSelectionBehavior(QAbstractItemView::SelectRows);

    // === PERBAIKAN TABEL ===
    tableJadwal->setEditTriggers(QAbstractItemView::NoEditTriggers);     // MATIKAN EDIT
    tableJadwal->verticalHeader()->setVisible(false);                    // HILANGKAN NOMOR BARIS
    tableJadwal->setShowGrid(false);                                     // Hilangkan garis grid
    tableJadwal->setAlternatingRowColors(false);


    // Info Panel Bawah
    QHBoxLayout *infoLayout = new QHBoxLayout();
    lblTotalSks = new QLabel("Total SKS: 0 / 20", this);
    lblTotalSks->setObjectName("SksLabel");

    lblStatusBentrok = new QLabel("Status: Aman", this);
    lblStatusBentrok->setObjectName("StatusAman");

    infoLayout->addWidget(lblTotalSks);
    infoLayout->addStretch();
    infoLayout->addWidget(lblStatusBentrok);

    // Tombol Aksi Kanan
    QHBoxLayout *actionLayout = new QHBoxLayout();
    btnHapus = new QPushButton(" Hapus Matkul", this);
    btnHapus->setObjectName("BtnHapus");
    btnSimpan = new QPushButton(" Simpan CSV", this);
    btnSimpan->setObjectName("BtnSimpan");

    actionLayout->addWidget(btnHapus);
    actionLayout->addWidget(btnSimpan);

    rightLayout->addWidget(lblJadwal);
    rightLayout->addWidget(tableJadwal);
    rightLayout->addLayout(infoLayout);
    rightLayout->addLayout(actionLayout);

    // Gabungkan ke layout utama
    mainLayout->addLayout(leftLayout, 2);
    mainLayout->addLayout(rightLayout, 3);

    // Connect Signal & Slot
    connect(btnTambah, &QPushButton::clicked, this, &MainWindow::tambahMatkul);
    connect(btnHapus, &QPushButton::clicked, this, &MainWindow::hapusMatkul);
    connect(btnSimpan, &QPushButton::clicked, this, &MainWindow::simpanKeCSV);
}

bool MainWindow::cekBentrok(const MataKuliah& m1, const MataKuliah& m2) {
    if (m1.hari != m2.hari) return false;
    // Bentrok jika jam mulai m1 berada di antara jam m2, atau sebaliknya
    return (m1.jamMulai < m2.jamSelesai() && m1.jamSelesai() > m2.jamMulai);
}

void MainWindow::tambahMatkul() {
    int r = listMasterMatkul->currentRow();
    if (r < 0) return;

    MataKuliah dipilih = masterMatkul[r];

    // Cek apakah sudah pernah ditambah sebelumnya
    // Cari bagian ini di fungsi tambahMatkul(), lalu tambahkan std::as_const
    for (const auto& m : std::as_const(terpilihMatkul)) {
        if (m.kode == dipilih.kode && m.nama == dipilih.nama) {
            QMessageBox::warning(this, "Peringatan", "Mata kuliah ini sudah diambil!");
            return;
        }
    }

    terpilihMatkul.append(dipilih);
    updateJadwalTable();
}

void MainWindow::hapusMatkul() {
    int r = tableJadwal->currentRow();
    if (r < 0) {
        QMessageBox::warning(this, "Peringatan", "Pilih baris mata kuliah di tabel yang ingin dihapus!");
        return;
    }
    terpilihMatkul.removeAt(r);
    updateJadwalTable();
}

void MainWindow::updateJadwalTable() {
    tableJadwal->setRowCount(0);
    int totalSks = 0;
    bool adaBentrokGlobal = false;

    // Array penampung status bentrok per baris terindeks
    QVector<bool>isBentrokRow(terpilihMatkul.size(), false);

    // Algoritma Deteksi Bentrok (Komparasi Berpasangan)
    for (int i = 0; i < terpilihMatkul.size(); ++i) {
        for (int j = i + 1; j < terpilihMatkul.size(); ++j) {
            if (cekBentrok(terpilihMatkul[i], terpilihMatkul[j])) {
                isBentrokRow[i] = true;
                isBentrokRow[j] = true;
                adaBentrokGlobal = true;
            }
        }
    }

    // ... [Kode bagian atas updateJadwalTable tetap sama] ...

    // Render ke Tabel UI
    for (int i = 0; i < terpilihMatkul.size(); ++i) {
        const auto& m = terpilihMatkul[i];
        int row = tableJadwal->rowCount();
        tableJadwal->insertRow(row);

        tableJadwal->setItem(row, 0, new QTableWidgetItem(m.kode));
        tableJadwal->setItem(row, 1, new QTableWidgetItem(m.nama));
        tableJadwal->setItem(row, 2, new QTableWidgetItem(QString::number(m.sks)));

        QString waktuStr = QString("%1, %2-%3").arg(m.hari, m.jamMulai.toString("HH:mm"), m.jamSelesai().toString("HH:mm"));
        tableJadwal->setItem(row, 3, new QTableWidgetItem(waktuStr));
        tableJadwal->setItem(row, 4, new QTableWidgetItem(m.ruang));

        QTableWidgetItem *statusItem = new QTableWidgetItem();
        tableJadwal->setItem(row, 5, statusItem);

        if (isBentrokRow[i]) {
            statusItem->setText("⚠️ BENTROK");

            for(int col = 0; col < 6; ++col) {
                if(tableJadwal->item(row, col)) {
                    // Warna gelap kemerahan untuk background baris yang bentrok
                    tableJadwal->item(row, col)->setBackground(QColor(69, 26, 26));
                    // Warna teks pink/merah muda terang agar kontras dan terbaca
                    tableJadwal->item(row, col)->setForeground(QColor(254, 202, 202));
                }
            }
        } else {
            statusItem->setText("☑️ OK");

            for(int col = 0; col < 6; ++col) {
                if(tableJadwal->item(row, col)) {
                    // Kembalikan warna teks ke putih/abu terang jika jadwal aman
                    tableJadwal->item(row, col)->setForeground(QColor(241, 245, 249));
                }
            }
        }

        totalSks += m.sks;
    }


    // Update Banner Status Total SKS & Bentrok Global
    lblTotalSks->setText(QString("Total SKS: %1 / 20").arg(totalSks));
    if (adaBentrokGlobal) {
        lblStatusBentrok->setText("⚠️ Ada Jadwal Bentrok!");
        lblStatusBentrok->setObjectName("StatusBentrok");
    } else {
        lblStatusBentrok->setText("☑️ Jadwal Aman");
        lblStatusBentrok->setObjectName("StatusAman");
    }

    lblStatusBentrok->style()->unpolish(lblStatusBentrok);
    lblStatusBentrok->style()->polish(lblStatusBentrok);
}

void MainWindow::simpanKeCSV() {
    if (terpilihMatkul.isEmpty()) {
        QMessageBox::warning(this, "Ekspor Gagal", "Tidak ada jadwal untuk disimpan.");
        return;
    }

    QString fileName = QFileDialog::getSaveFileName(this, "Simpan Jadwal", "", "CSV Files (*.csv)");
    if (fileName.isEmpty()) return;

    QFile file(fileName);
    if (file.open(QIODevice::WriteOnly | QIODevice::Text)) {
        QTextStream out(&file);
        out << "Kode,Mata Kuliah,SKS,Hari,Jam Mulai,Jam Selesai,Ruang\n";
        // Menggunakan std::as_const
        for (const auto& m : std::as_const(terpilihMatkul)) {
            out << m.kode << "," << m.nama << "," << m.sks << "," << m.hari << ","
                << m.jamMulai.toString("HH:mm") << "," << m.jamSelesai().toString("HH:mm") << "," << m.ruang << "\n";
        }
        file.close();
        QMessageBox::information(this, "Sukses", "Jadwal berhasil disimpan ke CSV!");
    }
}
